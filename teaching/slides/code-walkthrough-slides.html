def call_llm(prompt, system_message="คุณกำลังเล่นเป็นตัวละคร ..."):
    """เรียก Claude API แล้วดึง JSON จาก response"""
    response = client.messages.create(
        model=MODEL,
        max_tokens=2048,
        system=system_message,        # บอก Claude ว่า "ทำตัวยังไง"
        messages=[{"role": "user", "content": prompt}]  # prompt จริง
    )
    text = response.content[0].text   # ดึงข้อความจาก response
Parameter ที่สำคัญ:
• system = กำหนด "บทบาท" ของ Claude (เช่น ให้เล่นเป็นตัวละคร)
• messages = prompt ที่ส่งไป (เป็น list ของ dict)
• max_tokens=2048 = จำกัดความยาว response
• response.content[0].text = ข้อความที่ Claude ตอบกลับมา

JSON Parsing — 3 Fallbacks

Claude อาจตอบ JSON ในหลายรูปแบบ — เราต้องรองรับทั้งหมด


# Fallback 1: ลอง parse ตรงๆ
try: return json.loads(text)
except json.JSONDecodeError: pass

# Fallback 2: หา JSON ใน code fence ```json ... ```
code_match = re.search(r'```(?:json)?\s*\n?([\s\S]*?)\n?```', text)
if code_match:
    try: return json.loads(code_match.group(1))
    except: pass

# Fallback 3: หาจาก { แรก ถึง } สุดท้าย
first, last = text.find('{'), text.rfind('}')
if first != -1 and last > first:
    try: return json.loads(text[first:last + 1])
    except: pass
Fallbackจับ patternตัวอย่าง
1. ตรงๆjson.loads(text){"action": "go"}
2. Code fenceregex หา ```json``````json {...} ```
3. Braces{ แรก ... } สุดท้ายSure! {"action":"go"}

@dataclass
class Memory:
    """ความทรงจำหนึ่งอันของ Generative Agent"""
    content: str                        # เนื้อหา เช่น "ไป Lotus's พนักงานดีมาก"
    created_at: datetime                # เวลาที่เกิดเหตุการณ์
    importance: int                     # 1-10 (LLM ให้คะแนน)
    memory_type: str = "observation"    # observation / reflection / plan
    related_store: Optional[str] = None # ร้านที่เกี่ยวข้อง (ถ้ามี)
    access_count: int = 0              # เรียกดูไปกี่ครั้ง

3 ประเภทความทรงจำ

reflection บทเรียนที่สรุปได้

plan แผนที่วางไว้

ตัวอย่าง

ref "Big C คุ้มแต่เสียเวลา"

MemoryStream class — __init__ + add()


class MemoryStream:
    """ระบบความทรงจำ — ค้นหาด้วย score = recency x importance x relevance"""

    RECENCY_DECAY = 0.99      # ค่าคงที่: ทุก 1 ชม. ความทรงจำจางลง 1%

    def __init__(self, agent_name: str):
        self.agent_name = agent_name
        self.memories: list[Memory] = []    # เก็บ Memory ทั้งหมดเป็น list

    def add(self, content, importance, memory_type="observation",
            created_at=None, related_store=None):
        mem = Memory(
            content=content,
            created_at=created_at or datetime.now(),
            importance=importance,
            memory_type=memory_type,
            related_store=related_store
        )
        self.memories.append(mem)
        return mem
โครงสร้างง่ายมาก:
• __init__: สร้าง list ว่าง เก็บชื่อ agent
• add(): สร้าง Memory object แล้ว append เข้า list
• RECENCY_DECAY = 0.99: ค่า decay ต่อชั่วโมง (จาก Stanford paper)

MemoryStream — retrieve() หัวใจของระบบ


def retrieve(self, query, top_k=5, current_time=None):
    if current_time is None:
        current_time = datetime.now()
    results = []
    for mem in self.memories:
        # 1. Recency: ยิ่งเก่า ยิ่งคะแนนต่ำ
        hours = (current_time - mem.created_at).total_seconds() / 3600
        recency = self.RECENCY_DECAY ** hours        # 0.99^hours

        # 2. Importance: LLM ให้คะแนน 1-10 → normalize เป็น 0-1
        importance = mem.importance / 10.0

        # 3. Relevance: substring matching (รองรับภาษาไทย)
        query_keywords = set(query.lower().split())
        content_lower = mem.content.lower()
        matching = sum(1 for word in query_keywords if word in content_lower)
        relevance = min(1.0, matching / len(query_keywords)) if query_keywords else 0

        # คูณ 3 ตัวเข้าด้วยกัน
        total = recency * importance * relevance
        results.append((mem, {"recency": recency, "importance": importance,
                               "relevance": relevance, "total": total}))
    results.sort(key=lambda x: x[1]["total"], reverse=True)
    return results[:top_k]

สูตร Retrieval: score = recency x importance x relevance

ส่วนประกอบสูตรความหมาย
Recency 0.99 ^ hours ยิ่งเก่ายิ่งจาง — 24 ชม. = 0.79, 1 สัปดาห์ = 0.18
Importance importance / 10 LLM ให้คะแนน 1-10 — เหตุการณ์สำคัญจำได้นานกว่า
Relevance matching / total keywords ตรงกับ query มากแค่ไหน — ใช้ substring matching
ทำไมใช้ substring matching?
ภาษาไทยตัดคำยาก (ไม่มี space ระหว่างคำ)
ใช้ word in content_lower แทน exact match — เช่น query "นม" จะเจอ "ซื้อนมที่ Big C"
Paper ต้นฉบับใช้ vector embeddings ซึ่งแม่นกว่าแต่ซับซ้อนกว่ามาก

Cognitive Loop — 5 ขั้นตอน

วงจรการคิดของ Generative Agent

ภาพรวม: 5 Steps

1. PERCEIVE → 2. RETRIEVE → 3. PLAN → 4. EXECUTE → 5. REFLECT
↓
Memory Stream — REFLECT สร้าง memory ใหม่ใส่กลับ

def format_for_prompt(self, results):
    """แปลง memories เป็น text สำหรับใส่ใน LLM prompt"""
    if not results:
        return "ไม่มีความทรงจำที่เกี่ยวข้อง"
    return "\n".join(f"- {mem.content}" for mem, _ in results)

def build_prompt(agent, perception, memories_text):
    p = agent['personality']
    fin = agent['financial']
    return f"""คุณกำลังเล่นเป็นตัวละครชื่อ \"{agent['name']}\"

[ตัวตนของฉัน]
- อายุ {agent['age']} ปี อาชีพ {agent['job']}
- รายได้ {fin['monthly_income']:,} บาท/เดือน
- {agent['background']}

[นิสัยของฉัน (personality traits)]
- ความไวต่อราคา: {p['price_sensitivity']*10:.0f}/10
- ชอบโปรโมชั่น: {p['promotion_attraction']*10:.0f}/10
- ต้องการความสะดวก: {p['convenience_preference']*10:.0f}/10
- ซื้อของตามใจ: {p['impulse_buying']*10:.0f}/10
... (+ innate_traits, routines, pain_points)

[ความทรงจำที่เกี่ยวข้อง]         ← จาก RETRIEVE
{memories_text}

[สถานการณ์ + ของที่ต้องซื้อ + ร้านค้า]  ← จาก PERCEIVE
...

ตอบในรูปแบบ JSON: {{ "thinking_steps": [...], "decision": {{...}} }}"""
Prompt ประกอบด้วย 4 ส่วน: ตัวตน + นิสัย + ความทรงจำ + สถานการณ์
Claude ตัดสินใจ "ตามนิสัย" ของตัวละคร ไม่ใช่ตาม if-else

Key Takeaways

สรุปสิ่งที่ได้เรียนรู้

Patterns ที่สำคัญ

1. Memory Stream เป็นศูนย์กลาง

เก็บทุกประสบการณ์ ค้นหาด้วย recency x importance x relevance

2. LLM ใช้แค่ 2 จุด: PLAN + REFLECT

ที่เหลือเป็น Python ธรรมดา — ประหยัด API cost

3. Personality อยู่ใน Prompt ไม่ใช่ในโค้ด

เปลี่ยนนิสัยแค่เปลี่ยนตัวเลข — ไม่ต้องเขียน if-else ใหม่

4. REFLECT สร้างการเรียนรู้

Memory ใหม่จาก reflection ถูกใส่กลับ → ส่งผลต่อการตัดสินใจครั้งถัดไป

ทำไม LLM ดีกว่า if-else?

มิติ if-else (Rule-based) LLM (Generative Agent)
เปลี่ยน personality เขียน if-else ใหม่ทั้งหมด แก้ตัวเลขใน prompt
เพิ่มร้านค้าใหม่ เพิ่ม branch ในโค้ด เพิ่มใน STORES list
ความสมจริง ทำซ้ำเหมือนเดิมทุกครั้ง คิดจากบริบท ตอบต่างกันได้
การเรียนรู้ ไม่เรียนรู้ REFLECT สร้าง memory ใหม่
Scalability เพิ่ม agent = เพิ่ม code เพิ่มแค่ profile dict
Bottom line: LLM ทำให้สร้าง agent ใหม่ได้ง่าย
แค่กำหนด personality + memories — ไม่ต้องเขียนกฎตัดสินใจเลย
